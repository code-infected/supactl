-- Fetching recent active users with high credit scores
SELECT u.id, u.email, p.plan_type, p.created_at
FROM auth.users u
JOIN public.profiles p ON u.id = p.id
WHERE p.credits > 500 AND p.status IS NOT NULL
ORDER BY p.created_at DESC;
